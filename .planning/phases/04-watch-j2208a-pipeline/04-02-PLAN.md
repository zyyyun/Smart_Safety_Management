    ```python
    from typing import Optional

    HR_INVALID_MIN = 30
    HR_INVALID_MAX = 220
    TEMP_INVALID_MIN = 25.0
    TEMP_INVALID_MAX = 43.0
    DTEMP_NOISY_PER_SEC = 1.5  # |Δtemp| > 이 값 → NOISY

    @dataclass
    class Sample:
        ts: float                 # epoch seconds
        heart_rate: Optional[int] # parse_packet 의 heart_rate (없으면 None)
        temperature_c: Optional[float]

    def validate_sample(sample: Sample, prev: Optional[Sample]) -> dict:
        """per-field quality 분류. 반환 dict 키 = 'heart_rate', 'temperature_c'.
        값 ∈ {'GOOD', 'WARMUP', 'NOISY', 'INVALID'}.
        Per CONTEXT.md D-07."""
        q = {}
        # HR
        if sample.heart_rate is None:
            q["heart_rate"] = "INVALID"  # 데이터 없음
        elif sample.heart_rate == 0:
            q["heart_rate"] = "WARMUP"
        elif sample.heart_rate < HR_INVALID_MIN or sample.heart_rate > HR_INVALID_MAX:
            q["heart_rate"] = "INVALID"
        else:
            q["heart_rate"] = "GOOD"
        # temp
        if sample.temperature_c is None:
            q["temperature_c"] = "INVALID"
        elif sample.temperature_c < TEMP_INVALID_MIN or sample.temperature_c > TEMP_INVALID_MAX:
            q["temperature_c"] = "INVALID"
        elif prev is not None and prev.temperature_c is not None:
            dt = sample.ts - prev.ts
            if dt > 0:
                rate = abs(sample.temperature_c - prev.temperature_c) / dt
                if rate > DTEMP_NOISY_PER_SEC:
                    q["temperature_c"] = "NOISY"
                else:
                    q["temperature_c"] = "GOOD"
            else:
                q["temperature_c"] = "GOOD"
        else:
            q["temperature_c"] = "GOOD"
        return q
    ```

    ```python
    from collections import deque
    from dataclasses import dataclass, field
    from typing import Deque

    # Per CONTEXT.md D-06 — Python 상수 (v1.1 외부화)
    T_OFF  = 33.5     # °C, 케이스 외부 평형
    T_WARM = 35.5     # °C, 피부 밀착
    N1_OFF_ENTRY_SEC  = 30.0   # OFF 진입: HR=0 + temp≤T_OFF 가 N1 초 지속
    N2_WORN_ENTRY_SEC = 30.0   # WORN 진입: HR>0 안정 + temp≥T_WARM 가 N2 초 지속
    SLIDING_WINDOW_SEC = 5.0   # 5초 sliding window 다수결

    STATES = ("OFF", "WARMUP", "TRANSIENT", "WORN", "ABNORMAL")

    def _instantaneous_state(hr: int, temp: float) -> str:
        """1 sample 의 즉시 상태 — 진입 시간 누적은 caller 가 추적."""
        # ABNORMAL: 명백한 이상 — temp >43 또는 HR>220 등 (validate 후 거름)
        if hr is None or temp is None:
            return "TRANSIENT"
        if hr == 0 and temp <= T_OFF:
            return "OFF"
        if hr > 0 and temp >= T_WARM:
            return "WORN"
        # 그 외 = WARMUP (착용 중인데 PPG 락온 안 됨 등)
        return "WARMUP"

    @dataclass
    class StateMachine:
        window: Deque[tuple] = field(default_factory=lambda: deque())  # (ts, instant_state)

        def update(self, ts: float, hr: int, temp: float) -> str:
            inst = _instantaneous_state(hr, temp)
            self.window.append((ts, inst))
            # SLIDING_WINDOW_SEC 보다 오래된 항목 제거
            cutoff = ts - SLIDING_WINDOW_SEC
            while self.window and self.window[0][0] < cutoff:
                self.window.popleft()
            return self._majority()

        def _majority(self) -> str:
            if not self.window:
                return "TRANSIENT"
            counts = {}
            for _, s in self.window:
                counts[s] = counts.get(s, 0) + 1
            # 동률 시 TRANSIENT (전이 중 불안정)
            sorted_counts = sorted(counts.items(), key=lambda kv: -kv[1])
            if len(sorted_counts) > 1 and sorted_counts[0][1] == sorted_counts[1][1]:
                return "TRANSIENT"
            return sorted_counts[0][0]

    def classify_state(samples: list) -> str:
        """순수 함수 형태 — 테스트 편의용. samples = [(ts, hr, temp), ...]."""
        sm = StateMachine()
        last = "TRANSIENT"
        for ts, hr, temp in samples:
            last = sm.update(ts, hr, temp)
        return last
    ```

    **(B) `j2208a/aggregate.py`** — D-08 구현. statistics.median, IQR =
    Q3-Q1 (numpy 의존 회피, sorted slice 로 계산):
    ```python
    import statistics
    from dataclasses import dataclass
    from typing import Optional

    GOOD_RATIO_THRESHOLD = 0.5  # < 0.5 → NULL aggregates (per D-08)

    @dataclass
    class MinuteResult:
        hr_median: Optional[int]
        temp_median: Optional[float]
        temp_iqr: Optional[float]
        steps_delta: Optional[int]
        good_ratio: float
        dominant_state: Optional[str]

    def _iqr(values: list) -> float:
        if len(values) < 2:
            return 0.0
        s = sorted(values)
        n = len(s)
        q1 = s[n // 4]
        q3 = s[(3 * n) // 4]
        return q3 - q1

    def aggregate_minute(samples: list, dominant_state: Optional[str] = None) -> MinuteResult:
        """samples = [{'ts','hr','temp','steps','quality_hr','quality_temp'}, ...].
        Per CONTEXT.md D-08."""
        if not samples:
            return MinuteResult(None, None, None, None, 0.0, dominant_state)
        n = len(samples)
        good_hr  = [s["hr"]   for s in samples if s.get("quality_hr") == "GOOD"]
        good_tmp = [s["temp"] for s in samples if s.get("quality_temp") == "GOOD"]
        good_ratio_hr   = len(good_hr) / n if n else 0.0
        good_ratio_tmp  = len(good_tmp) / n if n else 0.0
        # 보수적 — HR/temp 둘 중 더 낮은 비율로 결측 판정
        good_ratio = min(good_ratio_hr, good_ratio_tmp)

        hr_median = int(statistics.median(good_hr)) if good_ratio_hr >= GOOD_RATIO_THRESHOLD else None
        temp_median = round(statistics.median(good_tmp), 1) if good_ratio_tmp >= GOOD_RATIO_THRESHOLD else None
        temp_iqr  = round(_iqr(good_tmp), 1) if good_ratio_tmp >= GOOD_RATIO_THRESHOLD else None

        # steps delta — 처음과 끝 sample 의 steps 차이
        steps_first = next((s["steps"] for s in samples if s.get("steps") is not None), None)
        steps_last  = next((s["steps"] for s in reversed(samples) if s.get("steps") is not None), None)
        steps_delta = (steps_last - steps_first) if (steps_first is not None and steps_last is not None) else None

        return MinuteResult(hr_median, temp_median, temp_iqr, steps_delta, good_ratio, dominant_state)
    ```

    ```python
    from dataclasses import dataclass, field
    from typing import Optional
    from enum import Enum

    # Per CONTEXT.md D-10 — testuser1 실제 나이로 갱신 가능
    TEST_USER_AGE = 30  # 운영자 보정 시 이 상수만 갱신

    WORN_GATE_SEC      = 60.0   # WORN ≥ 이 초 후에만 평가 (D-09)
    OFF_REMOVED_SEC    = 300.0  # OFF ≥ 5분 → 탈착 (D-09)
    COMMS_LOST_SEC     = 120.0  # raw 부재 ≥ 120s → 통신두절 (D-09)
    TACHY_KARVONEN     = 0.85   # HR ≥ (220-AGE) × 0.85 → 빈맥 (D-09)

    class Severity(str, Enum):
        NORMAL  = "NORMAL"   # 해소 알림용 (DB 의 severity CHECK 와 별도 — DB 는 CAUTION/WARNING/DANGER 만)
        CAUTION = "CAUTION"
        WARNING = "WARNING"
        DANGER  = "DANGER"

    ALERT_TYPES = ("TACHY", "REMOVED", "COMMS_LOST")

    @dataclass
    class AlertEvent:
        alert_type: str          # 'TACHY' | 'REMOVED' | 'COMMS_LOST'
        severity: str            # 'CAUTION' | 'WARNING' | 'DANGER' | 'NORMAL' (해소)
        is_resolution: bool      # True = 경보→정상 전이
        reason: dict

    @dataclass
    class DeriveContext:
        """프로세스 메모리 — 알림 전이 상태 (per advisor 가이드, DB 조회 X)."""
        last_severity: dict = field(default_factory=lambda: {
            "TACHY": "NORMAL", "REMOVED": "NORMAL", "COMMS_LOST": "NORMAL",
        })
        worn_since_ts: Optional[float] = None
        off_since_ts: Optional[float] = None

    def _karvonen_threshold(age: int) -> float:
        return (220 - age) * TACHY_KARVONEN

    def derive_alerts(
        ctx: DeriveContext,
        now_ts: float,
        wear_state: str,           # state_machine 결과
        hr_60s_median: Optional[int],
        last_raw_ts: Optional[float],
    ) -> list:
        """현재 상태에서 발생한 알림 (전이 시점만) 을 반환. Per D-09."""
        alerts = []

        # WORN 지속 시간 추적
        if wear_state == "WORN":
            if ctx.worn_since_ts is None:
                ctx.worn_since_ts = now_ts
        else:
            ctx.worn_since_ts = None

        # OFF 지속 시간 추적
        if wear_state == "OFF":
            if ctx.off_since_ts is None:
                ctx.off_since_ts = now_ts
        else:
            ctx.off_since_ts = None

        # WORN 60s 미만 → TACHY 평가만 skip (D-09 "WORN 60초 이상 지속 후에만
        # 평가" 는 빈맥 한정 — REMOVED 는 OFF 지속, COMMS_LOST 는 raw 부재이므로
        # WORN 게이트와 독립).
        worn_long_enough = (
            ctx.worn_since_ts is not None and (now_ts - ctx.worn_since_ts) >= WORN_GATE_SEC
        )

        # --- TACHY (WORN ≥ 60s 후에만) ---
        new_tachy_sev = "NORMAL"
        if worn_long_enough and hr_60s_median is not None:
            if hr_60s_median >= _karvonen_threshold(TEST_USER_AGE):
                new_tachy_sev = "WARNING"
        alerts.extend(_emit_transition(ctx, "TACHY", new_tachy_sev, now_ts,
                                        {"hr_60s_median": hr_60s_median,
                                         "threshold": _karvonen_threshold(TEST_USER_AGE)}))

        # --- REMOVED (OFF ≥ 5분) ---
        new_removed_sev = "NORMAL"
        if ctx.off_since_ts is not None and (now_ts - ctx.off_since_ts) >= OFF_REMOVED_SEC:
            new_removed_sev = "WARNING"
        alerts.extend(_emit_transition(ctx, "REMOVED", new_removed_sev, now_ts,
                                        {"off_duration_sec": (now_ts - ctx.off_since_ts) if ctx.off_since_ts else 0}))

        # --- COMMS_LOST (raw 부재 ≥ 120s) ---
        # last_raw_ts is None ≡ "raw 가 한 번도 도착하지 않음" (cold-start grace
        # period). 첫 raw 도착 전에는 COMMS_LOST 평가 자체를 skip — phantom
        # WARNING 으로 FCM 푸시되는 false-positive 차단. 첫 raw 도착 후에는
        # last_raw_ts 가 채워지고, 이후 ≥120s 부재 시 정상 발사.
        new_comms_sev = "NORMAL"
        if last_raw_ts is not None and (now_ts - last_raw_ts) >= COMMS_LOST_SEC:
            # WORN ≥ 60s 가 아니어도 통신두절은 경고 (BLE 자체 끊김)
            new_comms_sev = "WARNING"
        alerts.extend(_emit_transition(ctx, "COMMS_LOST", new_comms_sev, now_ts,
                                        {"last_raw_age_sec": (now_ts - last_raw_ts) if last_raw_ts else None}))

        return alerts

    def _emit_transition(ctx: DeriveContext, alert_type: str, new_sev: str,
                         now_ts: float, reason: dict) -> list:
        """전이 발생 시점에만 1건. 동일 등급 지속 = 0건."""
        prev = ctx.last_severity[alert_type]
        if new_sev == prev:
            return []
        # 전이 발생
        ctx.last_severity[alert_type] = new_sev
        is_resolution = (new_sev == "NORMAL")
        return [AlertEvent(
            alert_type=alert_type,
            severity=new_sev,
            is_resolution=is_resolution,
            reason=reason,
        )]
    ```
