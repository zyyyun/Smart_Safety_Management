    ```sql
    -- ============================================================
    -- J2208A 워치 24h 운용 검증 SQL (per CONTEXT.md D-13)
    --
    -- 사용법:
    --   psql $DATABASE_URL -v device_id=N -f scripts/verify_watch_24h.sql
    --   (Supabase SQL Editor 에서 실행 시 :'device_id' 를 실제 값으로 치환)
    -- ============================================================

    \set device_id 1   -- 04-01 시드의 testuser1 디바이스 device_id 로 갱신

    -- ── §1. raw_events 24h 카운트 (목표: dedup 후 ≥ 80,000) ──
    SELECT 'raw_events_24h_count' AS metric,
           count(*) AS value
      FROM public.raw_events
     WHERE device_id = :device_id
       AND ts >= now() - interval '24 hours';

    -- ── §2. minute_summary 24h 카운트 (목표: 1440 ± 결측) ──
    SELECT 'minute_summary_24h_count' AS metric,
           count(*) AS value
      FROM public.minute_summary
     WHERE device_id = :device_id
       AND minute_ts >= now() - interval '24 hours';

    -- ── §3. minute_summary good_ratio 분포 (NULL 집계 비율) ──
    SELECT
        count(*) FILTER (WHERE good_ratio >= 0.5)            AS minutes_good,
        count(*) FILTER (WHERE good_ratio <  0.5)            AS minutes_missing,
        count(*) FILTER (WHERE good_ratio IS NULL)           AS minutes_null,
        round(100.0 * count(*) FILTER (WHERE good_ratio >= 0.5)
              / NULLIF(count(*), 0), 1)                       AS good_pct
      FROM public.minute_summary
     WHERE device_id = :device_id
       AND minute_ts >= now() - interval '24 hours';

    -- ── §4. wear_state_events 24h 전이 내역 ──
    SELECT ts, from_state, to_state, reason
      FROM public.wear_state_events
     WHERE device_id = :device_id
       AND ts >= now() - interval '24 hours'
     ORDER BY ts;

    -- ── §5. safety_alerts 24h 발생 내역 ──
    SELECT alert_id, alert_type, severity, raised_at, resolved_at, reason
      FROM public.safety_alerts
     WHERE device_id = :device_id
       AND raised_at >= now() - interval '24 hours'
     ORDER BY raised_at;

    -- ── §6. (참고) 1분 윈도우 비어있는 슬롯 검출 ──
    -- 실제 minute_ts 시계열 vs 24h × 60분 = 1440 슬롯 비교
    WITH expected AS (
        SELECT generate_series(
            date_trunc('minute', now() - interval '24 hours'),
            date_trunc('minute', now()),
            interval '1 minute'
        ) AS minute_ts
    )
    SELECT count(*) AS missing_minutes
      FROM expected e
      LEFT JOIN public.minute_summary m
        ON m.device_id = :device_id AND m.minute_ts = e.minute_ts
     WHERE m.minute_ts IS NULL;

    -- ── §7. raw 수신율 추정 (5–6Hz 기준) ──
    -- 5.5 Hz × 86400s × 0.95 ≈ 451440 raw notify 기대치 (dedup 전).
    -- dedup 후 1초 단위 ≈ 86400 × 0.95 ≈ 82080 행 기대.
    SELECT
        count(*)                                                     AS raw_rows_24h,
        round(100.0 * count(*) / 82080.0, 1)                         AS raw_dedup_pct_of_expected;
    ```

    ```sql
    --   §1: ≥ 80,000  (실 수신율 ≥ 95% × dedup 후)
    --   §2: 1440 ± 결측 (행 자체 누락은 0 이어야 함, good_ratio<0.5 인 행은 허용)
    --   §6: missing_minutes = 0 — 분 행 자체가 빠지면 phase 실패
    --   §3: minutes_good 비율 ≥ 90% (운영자 판정 대비 — 식사·샤워 시 OFF/WARMUP 정상)
    --   §4: 운영자가 의도한 전이 (착용·탈착·식사) 와 일치
    --   §5: 의도된 5분 탈착 시나리오에서 REMOVED 1행 + resolved_at 채워짐
    ```
