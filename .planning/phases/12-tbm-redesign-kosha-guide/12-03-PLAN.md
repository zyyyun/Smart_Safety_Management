       package com.example.smart_safety_management.tbm

       /**
        * Compose state 의 hazards/controls List 의 immutable add/edit/delete reducer.
        * Pitfall P-NEW-3 mitigation — recomposition trigger 를 위해 새 List instance 반환 강제.
        * 신규 추가 항목은 is_custom=true 태그 (D-01).
        */
       object HazardsListReducer {
         fun add(current: List<TbmTemplateHazard>, text: String): List<TbmTemplateHazard> {
           val newId = generateNextId(current.map { it.id })
           // dup id 방지 — generateNextId 가 항상 unique
           return current + TbmTemplateHazard(id = newId, text = text, isCustom = true)
         }

         fun edit(current: List<TbmTemplateHazard>, id: String, newText: String): List<TbmTemplateHazard> =
           current.map { if (it.id == id) it.copy(text = newText) else it }

         fun delete(current: List<TbmTemplateHazard>, id: String): List<TbmTemplateHazard> =
           current.filterNot { it.id == id }

         internal fun generateNextId(existingIds: List<String>): String {
           // 패턴: "h1", "h2", ..., "h99" — 숫자 suffix 추출, max+1
           val maxNum = existingIds
             .mapNotNull { it.removePrefix("h").toIntOrNull() }
             .maxOrNull() ?: 0
           return "h${maxNum + 1}"
         }
       }
